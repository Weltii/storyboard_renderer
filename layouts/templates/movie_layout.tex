% Commands to generate parts of the storyboard
\newcommand{\storyboardFrameHead}[4]{
	\vspace{1cm}

	Scene: #1 \hspace{.5cm} Einstellung: #2 \hspace{.5cm} Abschnitt: #3
	\hfill
	TC: #4 % timecode
}

\newcommand{\storyboardFrameContent}[3]{
	\vspace{.5cm}

	\begin{tabular}{c p{.3\textwidth} p{.3\textwidth}}
		\begin{minipage}[t]{.3\textwidth}
			\vspace{-\ht\strutbox}\includegraphics[width=\textwidth]{#1}
		\end{minipage} &
		\begin{minipage}[t]{.3\textwidth}
			\textbf{Bild} \\
		#2
		\end{minipage} &
		\begin{minipage}[t]{.3\textwidth}
			\textbf{Audio} \\
			#3
		\end{minipage}
	\end{tabular}
}

\newcommand{\storyboardFrameFooter}[6]{
	% one blank line is requierd!

	\vspace{.5cm}
	EG: \textbf{#1} \hspace{\fill}
	P: \textbf{#2} \hspace{\fill}
	Kamera: \textbf{#3} mm - \textbf{#4} - \textbf{#5} \hspace{\fill}
	FX: \textbf{#6} \hspace{2cm}
}
